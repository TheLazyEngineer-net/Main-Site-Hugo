{{- $icon := . -}} {{- $svgPath := "" -}} {{- if or (strings.HasSuffix $icon ".svg") (strings.HasPrefix $icon "svg/") }} {{- $svgPath = $icon -}} {{- else }} {{- $svgPath = printf "svg/%s.svg" $icon -}} {{- end }} {{- $svg := resources.Get $svgPath -}} {{- if not $svg }} {{- errorf "Failed to get icon: %v (resolved path: %s)\n\n" $icon $svgPath }} {{- end }} {{- if not $svg.Content }} {{- errorf "Empty content for icon: %v (resolved path: %s)\n\n" $icon $svgPath }} {{- end }} {{- $svg.Content | safeHTML -}}
